# Swarmer — Phase 3: Secrets Management

**Date:** 2026-04-14
**Status:** Implemented
**Depends on:** Phase 2 (Workspaces)

## Goal

Per-workspace secrets in two categories:
1. **OpenCode secret** — GCP/Vertex AI + Google AI Studio (Gemini) creds. One per workspace. Synced to `opencode-secret` K8s Secret.
2. **GitHub PATs** — HTTPS git authentication tokens. Multiple per workspace. Synced to `github-pat-<slug>` K8s Secrets.

Both stored Fernet-encrypted in SQLite; synced to Kubernetes Secrets immediately on save.

## Database Schema

```sql
CREATE TABLE opencode_secrets (
    id                                  INTEGER PRIMARY KEY AUTOINCREMENT,
    workspace_id                        INTEGER NOT NULL UNIQUE
                                            REFERENCES workspaces(id) ON DELETE CASCADE,
    google_cloud_project                TEXT NOT NULL DEFAULT '',
    vertex_location                     TEXT NOT NULL DEFAULT '',
    application_default_credentials_enc TEXT NOT NULL DEFAULT '',  -- Fernet(ADC JSON)
    google_api_key_enc                  TEXT NOT NULL DEFAULT '',  -- Fernet(key)
    created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE github_pats (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    workspace_id    INTEGER NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    name            TEXT NOT NULL,          -- user-defined label e.g. "jpacker-personal"
    github_username TEXT NOT NULL,
    pat_enc         TEXT NOT NULL,          -- Fernet(PAT value)
    description     TEXT NOT NULL DEFAULT '',
    created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (workspace_id, name)
);
```

### Encrypted Column Convention

Columns ending in `_enc` store Fernet ciphertext. Python `@property` setters call `crypto.encrypt()`; getters call `crypto.decrypt()`. Raw ciphertext is never sent to the browser.

## Models

### `swarmer/models/opencode_secret.py`

```python
class OpencodeSecret(Base):
    __tablename__ = "opencode_secrets"
    id:                         Mapped[int] = mapped_column(primary_key=True)
    workspace_id:               Mapped[int] = mapped_column(ForeignKey("workspaces.id"), unique=True)
    google_cloud_project:       Mapped[str] = mapped_column(Text, default="")
    vertex_location:            Mapped[str] = mapped_column(Text, default="")
    application_default_credentials_enc: Mapped[str] = mapped_column(Text, default="")
    google_api_key_enc:         Mapped[str] = mapped_column(Text, default="")

    @property
    def application_default_credentials(self) -> str:
        return crypto.decrypt(self.application_default_credentials_enc) \
               if self.application_default_credentials_enc else ""

    @application_default_credentials.setter
    def application_default_credentials(self, value: str):
        self.application_default_credentials_enc = crypto.encrypt(value) if value else ""

    @property
    def google_api_key(self) -> str:
        return crypto.decrypt(self.google_api_key_enc) if self.google_api_key_enc else ""

    @google_api_key.setter
    def google_api_key(self, value: str):
        self.google_api_key_enc = crypto.encrypt(value) if value else ""

    @property
    def has_adc(self) -> bool:
        return bool(self.application_default_credentials_enc)

    @property
    def masked_api_key(self) -> str:
        if not self.google_api_key_enc:
            return ""
        key = self.google_api_key
        return key[:4] + "..." + key[-4:] if len(key) > 8 else "****"
```

### `swarmer/models/github_pat.py`

```python
class GitHubPAT(Base):
    __tablename__ = "github_pats"
    id:              Mapped[int]      = mapped_column(primary_key=True)
    workspace_id:    Mapped[int]      = mapped_column(ForeignKey("workspaces.id"))
    name:            Mapped[str]      = mapped_column(Text)
    github_username: Mapped[str]      = mapped_column(Text)
    pat_enc:         Mapped[str]      = mapped_column(Text)
    description:     Mapped[str]      = mapped_column(Text, default="")

    @property
    def pat(self) -> str:
        return crypto.decrypt(self.pat_enc)

    @pat.setter
    def pat(self, value: str):
        self.pat_enc = crypto.encrypt(value)

    @property
    def k8s_secret_name(self) -> str:
        """Returns e.g. 'github-pat-jpacker-personal' from name 'jpacker-personal'."""
        slug = re.sub(r"[^a-z0-9]+", "-", self.name.lower()).strip("-")
        return f"github-pat-{slug}"
```

## K8s Secret Sync (`swarmer/k8s.py`)

```python
def apply_opencode_secret(namespace: str, secret: OpencodeSecret) -> None:
    """Create or replace `opencode-secret` K8s Secret."""
    data = {
        "GOOGLE_CLOUD_PROJECT": _b64(secret.google_cloud_project),
        "VERTEX_LOCATION":      _b64(secret.vertex_location),
        "GOOGLE_API_KEY":       _b64(secret.google_api_key),
    }
    if secret.has_adc:
        data["application_default_credentials.json"] = _b64(
            secret.application_default_credentials
        )
    _apply_secret(namespace, "opencode-secret", data)

def apply_github_pat_secret(namespace: str, pat: GitHubPAT) -> None:
    """Create or replace `github-pat-<slug>` K8s Secret."""
    _apply_secret(namespace, pat.k8s_secret_name, {
        "GITHUB_PAT":      _b64(pat.pat),
        "GITHUB_USERNAME": _b64(pat.github_username),
    })

def delete_github_pat_secret(namespace: str, pat: GitHubPAT) -> None:
    _delete_secret(namespace, pat.k8s_secret_name)

def _apply_secret(namespace: str, name: str, data: dict[str, str]) -> None:
    """replace → 404 → create pattern."""
    body = client.V1Secret(
        metadata=client.V1ObjectMeta(name=name, namespace=namespace),
        type="Opaque",
        data=data,
    )
    try:
        v1.replace_namespaced_secret(name, namespace, body)
    except client.exceptions.ApiException as exc:
        if exc.status == 404:
            v1.create_namespaced_secret(namespace, body)
        else:
            raise
```

## Templates

### `secrets/opencode_form.html`
- `enctype="multipart/form-data"` for ADC file upload
- Two sections: Claude/Vertex (project, location, ADC upload) and Gemini (API key password input + masked display)
- PAT list card below: table of PATs with Edit and Delete actions; "Add PAT" button

### `secrets/github_pat_form.html`
- Fields: Name, GitHub Username, PAT (password; placeholder "Leave blank to keep" on edit), Description
- PAT value is **never** echoed back to the browser

## Security Constraints

- PAT values never sent to browser (write-only from UI perspective)
- ADC JSON content never sent to browser (only "Uploaded ✓" indicator)
- `crypto.decrypt()` called only when constructing K8s secret payloads
- K8s secrets applied immediately on save — cluster stays in sync with DB
- Workspace deletion cascades: all PAT K8s secrets are removed when the namespace is deleted (namespace deletion removes all its secrets)

## Pod Usage of Secrets

Session pods consume these secrets as follows:

```python
# Main container: envFrom pulls all keys from opencode-secret
env_from = [
    client.V1EnvFromSource(
        secret_ref=client.V1SecretEnvSource(name="opencode-secret", optional=True)
    )
]

# ADC credentials mounted as a file
volumes = [
    client.V1Volume(
        name="gcloud-creds",
        secret=client.V1SecretVolumeSource(
            secret_name="opencode-secret",
            items=[client.V1KeyToPath(
                key="application_default_credentials.json",
                path="credentials.json",
            )]
        ),
    ),
    ...
]
env = [
    client.V1EnvVar(
        name="GOOGLE_APPLICATION_CREDENTIALS",
        value="/app/gcloud/credentials.json",
    )
]

# GitHub PAT: injected into main container and init container (for git clone)
env.append(client.V1EnvVar(
    name="GITHUB_PAT",
    value_from=client.V1EnvVarSource(
        secret_key_ref=client.V1SecretKeySelector(
            name=pat.k8s_secret_name, key="GITHUB_PAT", optional=True
        )
    ),
))
env.append(client.V1EnvVar(
    name="GITHUB_USERNAME",
    value_from=...,  # same secret, key="GITHUB_USERNAME"
))
```
